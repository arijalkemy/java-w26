package com.mercadolibre.project_be_java_hisp_w26_team4.controller;

import com.mercadolibre.project_be_java_hisp_w26_team4.dtos.response.ProductLocationDto;
import com.mercadolibre.project_be_java_hisp_w26_team4.model.User;
import com.mercadolibre.project_be_java_hisp_w26_team4.service.IFreshProductService;
import com.mercadolibre.project_be_java_hisp_w26_team4.util.BatchOrder;
import java.util.List;
import lombok.RequiredArgsConstructor;
import org.springframework.http.ResponseEntity;
import org.springframework.security.core.annotation.AuthenticationPrincipal;
import org.springframework.validation.annotation.Validated;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;

@RestController
@RequestMapping("/api/v1/fresh_products")
@Validated
@RequiredArgsConstructor
public class FreshProductController {
    final IFreshProductService freshProductService;

    @GetMapping("/{prodId}/batch/list")
    public ResponseEntity<List<ProductLocationDto>> getProductLocationsByType(
            @PathVariable Long prodId,
            @AuthenticationPrincipal User user,
            @RequestParam(defaultValue = "L") BatchOrder order
    ) {
        return ResponseEntity
                .ok(freshProductService.getProductLocationsByType(user, order, prodId));
    }

}
